-------- Profile Dataset test
# Define the JSON payload as a PowerShell object
$body = @{
    file_path    = "test_data.csv"
    dataset_name = "Test Data"
} | ConvertTo-Json

# Send the request
Invoke-RestMethod -Uri "http://localhost:8001/profile_dataset" `
                  -Method Post `
                  -ContentType "application/json" `
                  -Body $body
